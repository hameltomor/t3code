import { ServiceMap } from "effect"
import type { Effect } from "effect"
import type { HistoryImportParseError } from "../Errors.ts"

/** A parsed message extracted from a Codex rollout */
export interface ParsedCodexMessage {
  readonly role: "user" | "assistant" | "system"
  readonly text: string
  readonly createdAt: string // ISO datetime
  readonly turnId: string | null
  readonly isStreaming: boolean // true if message was interrupted mid-stream
}

/** A parsed activity extracted from a Codex rollout */
export interface ParsedCodexActivity {
  readonly kind: string // "exec_command" | "approval_request" | "patch_approval" | "error" | "compaction" | ...
  readonly summary: string
  readonly tone: "info" | "tool" | "approval" | "error"
  readonly turnId: string | null
  readonly createdAt: string
  readonly payload: unknown
}

/** Full parse result from a Codex rollout file */
export interface CodexRolloutParseResult {
  readonly sessionId: string | null
  readonly sessionMeta: {
    readonly cwd: string | null
    readonly model: string | null
    readonly source: unknown
    readonly cliVersion: string | null
    readonly gitBranch: string | null
    readonly gitCommit: string | null
  } | null
  readonly messages: ReadonlyArray<ParsedCodexMessage>
  readonly activities: ReadonlyArray<ParsedCodexActivity>
  readonly warnings: ReadonlyArray<string>
  readonly totalLinesProcessed: number
  readonly compactionCount: number
}

export interface CodexRolloutParserShape {
  /**
   * Parse a Codex rollout file, streaming line-by-line.

    options?: {

**2. Create `apps/server/src/historyImport/Layers/CodexRolloutParser.ts`:**

Implement the streaming parser. This is the most complex piece -- handle each case carefully.

**2a. File streaming setup:**
Use Effect FileSystem for streaming:
```typescript
import { FileSystem } from "effect"

const streamLines = (filePath: string) =>
  Effect.gen(function* () {
    const fs = yield* FileSystem.FileSystem
    return fs.stream(filePath, { chunkSize: FileSystem.KiB(64) }).pipe(
      Stream.decodeText(),
      Stream.splitLines,
      Stream.filter((line) => line.trim().length > 0),
    )
  }).pipe(Stream.unwrap)
```

If `FileSystem.FileSystem` is not available in this Effect version, fall back to `node:fs` with `createReadStream` + `node:readline` wrapped in `Effect.tryPromise`. The key requirement is: NEVER load the full file into memory (NFR-1).

**2b. Line parsing with tolerance:**
For each line, try `JSON.parse` then validate against `CodexRolloutLine` schema:
```typescript
Stream.map((line) => {
  try {
    const parsed = JSON.parse(line)
    // Use Schema.decodeUnknownEither for tolerant parsing
    const result = Schema.decodeUnknownEither(CodexRolloutLine)(parsed)
    return Either.isRight(result) ? Either.right(result.right) : Either.left(line)
  } catch {
    return Either.left(line) // malformed JSON, skip
  }
})
```

Skip malformed lines (Pitfall 5: concurrent write producing truncated last line). Add a warning for each skipped line.

**2c. Stateful accumulation with compaction reset:**
Use `Stream.runFold` or `Stream.mapAccum` to accumulate state as lines are processed:

```typescript
interface ParserState {
  sessionId: string | null
  sessionMeta: CodexRolloutParseResult["sessionMeta"]
  messages: ParsedCodexMessage[]
  activities: ParsedCodexActivity[]
  warnings: string[]
  currentTurnId: string | null
  currentTurnModel: string | null
  linesProcessed: number
  compactionCount: number
  messageCapped: boolean
  activityCapped: boolean
}
```

For each `CodexRolloutLine`, dispatch on `line.type`:

**`"session_meta"`:**
- Decode payload with `CodexSessionMetaLine` schema
- Extract `sessionId = meta.id`, `cwd = meta.cwd`, `model = meta.model_provider`
- Check if subagent via `isSubAgentSession(meta)` -- if so, add warning and note in result but continue parsing (the scanner already filtered subagents, this is defense in depth)

**`"response_item"`:**
- Decode payload with `CodexResponseItem` schema
- If `type === "message"` and `role === "user"`:
  - Extract text from `content` array (join all `text` fields where `type === "input_text"`)
  - Add as `ParsedCodexMessage` with `role: "user"`
- If `type === "message"` and `role === "assistant"`:
  - Extract text from `content` array (join all `text` fields where `type === "output_text"`)
  - Add as `ParsedCodexMessage` with `role: "assistant"`, `isStreaming: false`
- If `type === "reasoning"`:
  - If `encrypted_content` is present: SKIP entirely (Pitfall 4), add warning "Skipped encrypted reasoning"
  - If `summary` is present: Extract summary text, add as activity with `kind: "reasoning"`, `tone: "info"`
- If `type === "local_shell_call"` or `"function_call"`:
  - Add as activity with `kind: item.type`, `tone: "tool"`
- If `type === "function_call_output"`:
  - Add as activity with `kind: "function_call_output"`, `tone: "tool"`

**`"compacted"`:** (Pitfall 1 -- CRITICAL)
- Decode payload with `CodexCompactedItem` schema
- **DISCARD ALL accumulated messages and activities** (this is the compaction reset)
- Increment `compactionCount`
- If `replacement_history` is present and non-empty:
  - Process each `ResponseItem` in `replacement_history` as if it were a `response_item` line (reuse the same processing logic)
- If `replacement_history` is absent or empty:
  - Create a single system message from `compacted.message`: `{ role: "system", text: "[Context compacted] " + compacted.message }`
- Add warning: `"Context compaction detected -- pre-compaction messages discarded"`

**`"turn_context"`:**
- Decode payload with `CodexTurnContext` schema
- Update `currentTurnId = payload.turn_id`
- Update `currentTurnModel = payload.model`

**`"event_msg"`:**
- Decode payload with `CodexEventMsg` schema
- Dispatch on `payload.type`:
  - `"user_message"`: Add as `ParsedCodexMessage` with `role: "user"`, text from `payload.message`
  - `"agent_message"`: Add as `ParsedCodexMessage` with `role: "assistant"`, text from `payload.message`
  - `"task_started"` / `"turn_started"`: Update `currentTurnId`
  - `"task_complete"` / `"turn_complete"`: Add activity `kind: "turn_complete"`, `tone: "info"`
  - `"exec_command_begin"`: Add activity `kind: "exec_command"`, `tone: "tool"`, summary includes command and cwd
  - `"exec_command_end"`: Add activity `kind: "exec_command_end"`, `tone: exit_code === 0 ? "info" : "error"`
  - `"exec_approval_request"` / `"apply_patch_approval_request"`: Add activity `kind: "approval_request"`, `tone: "approval"`
  - `"error"`: Add activity `kind: "error"`, `tone: "error"`, summary = payload.message
  - `"context_compacted"`: Treat same as `"compacted"` RolloutItem (reset messages/activities)
  - Other types: Skip silently (schema tolerance)

**2d. Message capping for preview:**
When `options.maxMessages` is set, stop accumulating messages after the cap is reached (set `messageCapped = true`). Continue processing the file for activities and metadata, but don't add more messages. Same for `options.maxActivities`.

**2e. Post-processing -- force-complete streaming messages (Pitfall 8):**
After all lines are processed, scan `messages` array. Any message where we detect it was the last assistant message and the file ended without a completion marker should have `isStreaming` set to `false`. The simplest heuristic: if the last message is `role: "assistant"`, mark it `isStreaming: false` regardless (imports are always complete).

**2f. Timestamp handling:**
Use the `timestamp` field from each `CodexRolloutLine` as `createdAt` for messages and activities. If `timestamp` is missing, use the current time.

**2g. Return result:**
Collect final state into `CodexRolloutParseResult` and return.

**Dependencies:**
- `FileSystem.FileSystem` from Effect (if available) or `node:fs` + `node:readline`
- Codex schemas from `../Schemas/CodexRolloutSchemas.ts`

**2a. `list` implementation:**
```typescript
const list = (input: HistoryImportListInput) => Effect.gen(function* () {
  const scanner = yield* CodexHistoryScannerService
  const catalogRepo = yield* HistoryImportCatalogRepository

  // Determine workspace root from input
  const workspaceRoot = input.workspaceRoot

  // If no provider filter, or filter is "codex", scan Codex
  if (!input.providerFilter || input.providerFilter === "codex") {
    yield* scanner.scan({ workspaceRoot }).pipe(
      Effect.catchAll((scanError) =>
        // NFR-4: One provider failing must not block results from others
        Effect.logWarning("Codex scan failed", { error: scanError.message }).pipe(
          Effect.as([]),
        ),
      ),
    )
  }

  // Future: scan Claude Code, Gemini here (guarded by providerFilter)

  // Return catalog entries from database
  const entries = yield* catalogRepo.listByWorkspace({
    workspaceRoot,
    providerName: input.providerFilter,
  })

  // Map HistoryImportCatalogEntry to HistoryImportConversationSummary
  // The schemas are intentionally compatible -- catalog entry fields map 1:1 to summary
  return entries as unknown as ReadonlyArray<HistoryImportConversationSummary>
  // Note: If the types diverge, use Schema.decode here for proper mapping
})
```

Actually, the `HistoryImportCatalogEntry` from the persistence layer uses plain `Schema.String` for all fields, while `HistoryImportConversationSummary` from contracts uses branded types like `TrimmedNonEmptyString`. The mapping should use `Schema.decode(HistoryImportConversationSummary)` for each entry, or more practically, construct the summary objects explicitly since the field names are identical. Use `Schema.decodeUnknown(Schema.Array(HistoryImportConversationSummary))` on the raw entries array, mapping any decode errors to an `HistoryImportScanError`. Alternatively, since catalog entries are written by our own code and should always be valid, a simple cast with a comment explaining why it's safe is acceptable.

  const catalogRepo = yield* HistoryImportCatalogRepository
  const parser = yield* CodexRolloutParserService

  // Look up catalog entry

  // Actually, we need a getById method. Since the repository doesn't have one,

  // We need to add a simple helper or query. For now, use listByWorkspace with a filter

  //
  // Practical approach: Add a getByCatalogId method to the catalog repository
  // in this task (minor addition to existing file).

  const catalogEntry = yield* getCatalogEntryById(input.catalogId)
  if (!catalogEntry) {
    return yield* new HistoryImportNotFoundError({
      message: `Catalog entry not found: ${input.catalogId}`,
    })
  }

  // Parse the rollout file with caps
  const parseResult = yield* parser.parse(catalogEntry.sourcePath, {
    maxMessages,
    maxActivities: 20,
  })

  // Build preview response
  const preview: HistoryImportConversationPreview = {
    catalogId: catalogEntry.catalogId,
    providerName: catalogEntry.providerName as HistoryImportProvider,
    title: catalogEntry.title || "Untitled",
    messages: parseResult.messages.map((m) => ({
      role: m.role,
      text: m.text,
      createdAt: m.createdAt,
    })),
    activities: parseResult.activities.slice(0, 20).map((a) => ({
      kind: a.kind,
      summary: a.summary,
    })),
    totalMessageCount: parseResult.messages.length,
    totalActivityCount: parseResult.activities.length,
    isTruncated: parseResult.messages.length >= maxMessages,
    linkMode: catalogEntry.linkMode as HistoryImportLinkMode,
    warnings: parseResult.warnings,
  }

  return preview
})
